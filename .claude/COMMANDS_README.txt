━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ 생성된 커스텀 명령어:

  1. /go        → RED 단계 시작 (다음 테스트 작성)
  2. /red       → RED 단계 상세 진행
  3. /green     → GREEN 단계 (최소 코드로 통과)
  4. /refactor  → REFACTOR 단계 (코드 정리)
  5. /commit    → 변경사항 git 커밋
  6. /status    → 진행 상황 확인
  7. /help      → 전체 도움말 (이 문서와 유사)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🚀 빠른 시작:

  1. /go        ← 첫 테스트 시작
  2. /green     ← 테스트 통과
  3. /refactor  ← 코드 정리 (필요 시)
  4. /commit    ← 변경사항 저장
  5. 반복...

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📚 작업 흐름:

  /go (RED)
     ↓
  ❌ 테스트 실패 확인
     ↓
  /green (GREEN)
     ↓
  ✅ 테스트 통과 확인
     ↓
  /refactor (REFACTOR) [선택사항]
     ↓
  ✅ 테스트 계속 통과 확인
     ↓
  /commit (COMMIT)
     ↓
  /status (확인)
     ↓
  반복...

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎯 핵심 원칙:

  RED 단계:
    • 테스트는 반드시 실패해야 함
    • 프로덕션 코드 수정 금지
    • 테스트 코드만 작성

  GREEN 단계:
    • 가장 간단한 구현만
    • 중복은 무시
    • 리팩토링 금지

  REFACTOR 단계:
    • 한 번에 하나의 변경만
    • 각 단계 후 테스트 실행
    • 행동 변경 금지 (구조만 개선)

  COMMIT 단계:
    • 구조적 변경과 행동적 변경 분리
    • 구조적 변경을 먼저 커밋
    • 작고 자주 커밋
